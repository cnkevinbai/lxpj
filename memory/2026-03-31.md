### Vite 配置关键代码

```typescript
build: {
  outDir: 'dist',
  sourcemap: false,  // 禁用 sourcemap 减小体积
  chunkSizeWarningLimit: 1500,
  rollupOptions: {
    output: {
      manualChunks: (id) => {
        // React 核心
        if (id.includes('react/index') || id.includes('react-dom/index')) {
          return 'react-core';
        }
        if (id.includes('react-router-dom')) {
          return 'react-router';
        }
        // Ant Design 图标
        if (id.includes('@ant-design/icons')) {
          return 'antd-icons';
        }
        // Ant Design 按组件类型分割
        if (id.includes('antd/es/form/')) return 'antd-form';
        if (id.includes('antd/es/table/') || id.includes('antd/es/list/') || id.includes('antd/es/tree/')) return 'antd-data';
        if (id.includes('antd/es/layout/')) return 'antd-layout';
        if (id.includes('antd/es/menu/') || id.includes('antd/es/dropdown/')) return 'antd-nav';
        if (id.includes('antd/es/message/') || id.includes('antd/es/modal/') || id.includes('antd/es/notification/')) return 'antd-feedback';
        if (id.includes('antd/es/')) return 'antd-ui';
        // Pro Components
        if (id.includes('@ant-design/pro-table')) return 'pro-table';
        if (id.includes('@ant-design/pro-form')) return 'pro-form';
        if (id.includes('@ant-design/pro-layout')) return 'pro-layout';
        // 图表库（体积大，独立 chunk）
        if (id.includes('@ant-design/plots') || id.includes('@antv/')) {
          return 'plots-core';
        }
        // 工具库
        if (id.includes('dayjs')) return 'dayjs-vendor';
        if (id.includes('axios')) return 'axios-vendor';
```
