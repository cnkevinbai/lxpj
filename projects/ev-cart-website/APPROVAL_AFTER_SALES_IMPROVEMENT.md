## 🎯 核心算法

### 智能派单算法
```typescript
function selectBestEngineer(engineers, ticket) {
  return engineers
    .map(engineer => ({
      ...engineer,
      score: calculateScore(engineer, ticket)
    }))
    .sort((a, b) => b.score - a.score)[0]
}

function calculateScore(engineer, ticket) {
  let score = 0
  
  // 技能匹配（40 分）
  if (engineer.skills.includes(ticket.type)) {
    score += 40
  }
  
  // 距离远近（30 分）
  const distance = calculateDistance(engineer.location, ticket.customerLocation)
  score += Math.max(0, 30 - distance * 3)
  
  // 工作量（20 分）
  score += Math.max(0, 20 - engineer.currentTickets * 5)
  
  // 客户评分（10 分）
  score += engineer.rating * 2
  
  return score
}
```
