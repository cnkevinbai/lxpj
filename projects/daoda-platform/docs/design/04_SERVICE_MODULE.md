## 4.2 插件接口规范

```typescript
// 售后服务模块插件接口定义
interface ServiceModulePlugin {
  id: string;
  name: string;
  version: string;
  
  config: {
    enabled: boolean;
    required: boolean;
    dependencies: string[];
  };
  
  bootstrap(context: ServiceModuleContext): Promise<void>;
  destroy(): Promise<void>;
  routes: RouteConfig[];
  menus: MenuItem[];
  permissions: Permission[];
  events: { [eventName: string]: EventHandler };
}

// 工单事件钩子
interface WorkOrderHooks {
  // 工单创建前
  beforeCreate(workOrder: WorkOrderDTO): Promise<WorkOrderDTO>;
  // 工单创建后
  afterCreate(workOrder: WorkOrder): Promise<void>;
  // 工单分配前
  beforeAssign(workOrder: WorkOrder, assignee: User): Promise<void>;
  // 工单分配后
  afterAssign(workOrder: WorkOrder, assignee: User): Promise<void>;
  // 工单状态变更
  onStatusChange(workOrder: WorkOrder, oldStatus: string, newStatus: string): Promise<void>;
  // 工单完成
  onComplete(workOrder: WorkOrder): Promise<void>;
  // SLA预警
  onSLAWarning(workOrder: WorkOrder, remainingTime: number): Promise<void>;
  // SLA超时
  onSLATimeout(workOrder: WorkOrder): Promise<void>;
}

// 示例：现场服务模块
const FieldServicePlugin: ServiceModulePlugin = {
  id: 'service.field',
  name: '现场服务',
  version: '1.0.0',
  
  config: {
    enabled: true,
    required: false,
    dependencies: ['service.workorder']
  },
  
  async bootstrap(context) {
```

```typescript
            await handler(context);

            context.logger.error(`Scheduled task ${task.id} failed`, error);
          }

        this.scheduledTasks.set(task.id, job);
        context.logger.info(`Started scheduled task: ${task.name}`);
      }
    }
```

```typescript
/**
 * 智能派单引擎
 */
class DispatchEngine {
  private context: ModuleContext;
  private rules: DispatchRule[] = [];
  
  async initialize(): Promise<void> {
    // 加载派单规则
    this.rules = await this.loadRules();
  }
  
  async shutdown(): Promise<void> {
    this.rules = [];
  }
  
  /**
   * 智能派单
   */
  async autoDispatch(workOrderIds: string[], strategy: DispatchStrategy): Promise<DispatchResult[]> {
    const results: DispatchResult[] = [];
    
    for (const workOrderId of workOrderIds) {
      const workOrder = await this.getWorkOrder(workOrderId);
      const technicians = await this.getAvailableTechnicians(workOrder);
      
      // 根据策略选择最佳技术人员
      let selectedTechnician: Technician | null = null;
      
      switch (strategy) {
        case 'nearest':
          selectedTechnician = this.selectNearest(workOrder, technicians);
          break;
        case 'balanced':
          selectedTechnician = this.selectBalanced(workOrder, technicians);
          break;
        case 'skill_match':
          selectedTechnician = this.selectBySkill(workOrder, technicians);
          break;
      }
      
      if (selectedTechnician) {
        // 创建派单
        await this.createDispatch(workOrderId, selectedTechnician.id);
        results.push({
          workOrderId,
          technicianId: selectedTechnician.id,
          technicianName: selectedTechnician.name,
          success: true
        });
      } else {
        results.push({
          workOrderId,
          success: false,
          reason: 'No available technician'
        });
      }
    }
    
    return results;
  }
  
  /**
   * 选择最近的技术人员
   */
  private selectNearest(workOrder: WorkOrder, technicians: Technician[]): Technician | null {
    if (technicians.length === 0) return null;
    
    // 按距离排序
    const sorted = technicians.sort((a, b) => {
      const distA = this.calculateDistance(workOrder.location, a.location);
      const distB = this.calculateDistance(workOrder.location, b.location);
      return distA - distB;
    });
    
    return sorted[0];
  }
  
  /**
   * 选择负载均衡的技术人员
   */
  private selectBalanced(workOrder: WorkOrder, technicians: Technician[]): Technician | null {
    const idle = technicians.filter(t => t.status === 'idle');
    if (idle.length > 0) {
      return idle[Math.floor(Math.random() * idle.length)];
    }
    
    // 按当前任务数排序
    const sorted = technicians.sort((a, b) => a.todayTaskCount - b.todayTaskCount);
    return sorted[0];
  }
  
  /**
   * 选择技能匹配的技术人员
   */
  private selectBySkill(workOrder: WorkOrder, technicians: Technician[]): Technician | null {
    const requiredSkills = workOrder.requiredSkills || [];
    
    const matched = technicians.filter(t => {
      return requiredSkills.every(skill => t.skills.includes(skill));
    });
    
    if (matched.length === 0) return null;
    
    // 在匹配的技术人员中选择最近的
    return this.selectNearest(workOrder, matched);
  }
  
  private calculateDistance(loc1: Location, loc2: Location): number {
    // 计算两点间距离（简化实现）
    const lat1 = loc1.latitude;
    const lon1 = loc1.longitude;
    const lat2 = loc2.latitude;
    const lon2 = loc2.longitude;
    
    return Math.sqrt(
      Math.pow(lat1 - lat2, 2) + Math.pow(lon1 - lon2, 2)
    );
  }
}
```
