```typescript
// CRM模块插件接口定义
interface CRMModulePlugin {
  // 模块标识
  id: string;
  name: string;
  version: string;
  
  // 模块配置
  config: {
    enabled: boolean;
    required: boolean;
    dependencies: string[];
  };
  
  // 模块入口
  bootstrap(context: CRMModuleContext): Promise<void>;
  
  // 模块销毁
  destroy(): Promise<void>;
  
  // 模块路由
  routes: RouteConfig[];
  
  // 模块菜单
  menus: MenuItem[];
  
  // 模块权限
  permissions: Permission[];
  
  // 模块事件
  events: {
    [eventName: string]: EventHandler;
  };
}

// 模块上下文
interface CRMModuleContext {
  // 全局状态
  store: Store;
  
  // 路由实例
  router: Router;
  
  // 事件总线
  eventBus: EventBus;
  
  // API客户端
  api: APIClient;
  
  // 权限检查
  hasPermission: (permission: string) => boolean;
}

// 示例：线索管理模块
const LeadModulePlugin: CRMModulePlugin = {
  id: 'crm.lead',
  name: '线索管理',
  version: '1.0.0',
  
  config: {
    enabled: true,
    required: false,
    dependencies: ['crm.customer']
  },
  
  async bootstrap(context) {
    // 注册路由
    context.router.addRoutes(this.routes);
    
    // 订阅事件
    context.eventBus.on('customer.created', this.handleCustomerCreated);
  },
  
  async destroy() {
    // 清理资源
  },
  
  routes: [
    { path: '/lead', component: LeadList },
    { path: '/lead/:id', component: LeadDetail }
  ],
  
  menus: [
    { 
      id: 'lead', 
      name: '线索中心',

      children: [
        { id: 'lead-list', name: '线索列表', path: '/lead' },

      ]
    }
```

### 6.2.1 客户表 (customer)

```sql
CREATE TABLE customer (
  id VARCHAR(36) PRIMARY KEY,
  code VARCHAR(32) NOT NULL UNIQUE COMMENT '客户编号',
  name VARCHAR(200) NOT NULL COMMENT '客户名称',
  short_name VARCHAR(100) COMMENT '简称',
  type ENUM('enterprise', 'personal') DEFAULT 'enterprise' COMMENT '客户类型',
  level ENUM('A', 'B', 'C', 'D', 'VIP') DEFAULT 'C' COMMENT '客户等级',
  stage ENUM('potential', 'new', 'active', 'loyal', 'churned') DEFAULT 'potential' COMMENT '客户阶段',
  
  -- 行业信息
  industry VARCHAR(50) COMMENT '行业',
  sub_industry VARCHAR(50) COMMENT '子行业',
  
  -- 来源信息
  source VARCHAR(50) COMMENT '客户来源',
  source_detail VARCHAR(200) COMMENT '来源详情',
  
  -- 归属信息
  owner_id VARCHAR(36) COMMENT '负责人ID',
  department_id VARCHAR(36) COMMENT '所属部门ID',
  region_id VARCHAR(36) COMMENT '所属区域ID',
  
  -- 公海信息
  in_pool BOOLEAN DEFAULT FALSE COMMENT '是否在公海',
  pool_date DATE COMMENT '进入公海日期',
  pool_reason VARCHAR(200) COMMENT '进入公海原因',
  
  -- 统计信息
  total_amount DECIMAL(15,2) DEFAULT 0 COMMENT '累计成交金额',
  total_orders INT DEFAULT 0 COMMENT '累计订单数',
  last_order_date DATE COMMENT '最后成交日期',
  
  -- 备注信息
  remark TEXT COMMENT '备注',
  
  -- 审计字段
  created_by VARCHAR(36),
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  updated_by VARCHAR(36),
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  deleted_at DATETIME DEFAULT NULL COMMENT '软删除',
  
  INDEX idx_code (code),
  INDEX idx_name (name),
  INDEX idx_owner (owner_id),
  INDEX idx_stage (stage),
  INDEX idx_in_pool (in_pool),
  INDEX idx_created_at (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='客户表';
```

```typescript
/**
 * CRM模块主类
 */
import { BaseModule, ModuleContext, ModuleStatus } from '@daoda/core';

export class CRMModule extends BaseModule {
  readonly id = '@daoda/crm';
  readonly name = 'CRM客户关系管理';
  readonly version = '1.0.0';
  
  private subModules: Map<string, BaseModule> = new Map();
  private scheduledTasks: Map<string, NodeJS.Timeout> = new Map();
  
  protected _status: ModuleStatus = 'installed';
  
  async onInstall(context: ModuleContext): Promise<void> {
    context.logger.info(`Installing ${this.name}...`);
    
    await this.runMigrations('up');
    await this.initBaseData(context);
    await this.registerPermissions(context);
    
    context.logger.info(`${this.name} installed successfully`);
  }
  
  async onUninstall(context: ModuleContext): Promise<void> {
    context.logger.info(`Uninstalling ${this.name}...`);
    
    const hasData = await this.checkRelatedData(context);
    if (hasData) {
      throw new Error('模块存在关联数据，请先清理数据后再卸载');
    }
    
    await this.runMigrations('down');
    await this.unregisterPermissions(context);
    
    context.logger.info(`${this.name} uninstalled successfully`);
  }
  
  async onLoad(context: ModuleContext): Promise<void> {
    context.logger.info(`Loading ${this.name}...`);
    
    this.context = context;
    this._status = 'loaded';
    
    await this.registerServices(context);
    await this.loadConfig(context);
    await this.initSubModules(context);
    
    context.logger.info(`${this.name} loaded successfully`);
  }
  
  async onUnload(context: ModuleContext): Promise<void> {
    context.logger.info(`Unloading ${this.name}...`);
    
    await this.stopScheduledTasks();
    await this.unloadSubModules(context);
    context.serviceRegistry.unregisterAll(this.id);
    
    this._status = 'installed';
    context.logger.info(`${this.name} unloaded successfully`);
  }
  
  async onEnable(context: ModuleContext): Promise<void> {
    context.logger.info(`Enabling ${this.name}...`);
    
    await this.registerRoutes(context);
    await this.registerMenus(context);
    await this.startEventSubscriptions(context);
    await this.enableSubModules(context);
    await this.startScheduledTasks(context);
    
    this._status = 'running';
    context.eventBus.emit('module.enabled', { moduleId: this.id });
    
    context.logger.info(`${this.name} enabled successfully`);
  }
  
  async onDisable(context: ModuleContext): Promise<void> {
    context.logger.info(`Disabling ${this.name}...`);
    
    await this.stopScheduledTasks();
    await this.disableSubModules(context);
    
    context.router.unregisterAll(this.id);
    context.menuRegistry.unregisterAll(this.id);
    context.eventBus.clearModuleSubscriptions(this.id);
    
    this._status = 'disabled';
    context.eventBus.emit('module.disabled', { moduleId: this.id });
    
    context.logger.info(`${this.name} disabled successfully`);
  }
  
  async onBeforeUpdate(context: ModuleContext, newVersion: string): Promise<void> {
    await this.saveModuleState(context);
    await this.backupModuleData(context);
    
    if (this._status === 'running') {
      await this.onDisable(context);
    }
    await this.onUnload(context);
  }
  
  async onAfterUpdate(context: ModuleContext, oldVersion: string): Promise<void> {
    await this.runMigrations('up');
    await this.restoreModuleState(context);
    await this.onLoad(context);
    await this.onEnable(context);
  }
  
  private async registerServices(context: ModuleContext): Promise<void> {
    context.serviceRegistry.register('CustomerService', 
      new CustomerService(context), this.id);
    context.serviceRegistry.register('LeadService', 
      new LeadService(context), this.id);
    context.serviceRegistry.register('OpportunityService', 
      new OpportunityService(context), this.id);
    context.serviceRegistry.register('OrderService', 
      new OrderService(context), this.id);
  }
  
  private async startScheduledTasks(context: ModuleContext): Promise<void> {
    // 客户阶段更新任务
    const stageUpdateTask = context.scheduler.schedule('0 1 * * *', async () => {
      const customerService = context.serviceRegistry.get<CustomerService>('CustomerService');
      await customerService.updateCustomerStages();
    });
    this.scheduledTasks.set('customer-stage-update', stageUpdateTask);
    
    // 商机预测任务
    const forecastTask = context.scheduler.schedule('0 6 * * *', async () => {
      const opportunityService = context.serviceRegistry.get<OpportunityService>('OpportunityService');
      await opportunityService.calculateForecast();
    });
    this.scheduledTasks.set('opportunity-forecast', forecastTask);
  }
  
  private async stopScheduledTasks(): Promise<void> {
    for (const [id, task] of this.scheduledTasks) {
      task.stop();
    }
    this.scheduledTasks.clear();
  }
  
  // 子模块管理方法
  private async initSubModules(context: ModuleContext): Promise<void> { /* ... */ }
  private async enableSubModules(context: ModuleContext): Promise<void> { /* ... */ }
  private async disableSubModules(context: ModuleContext): Promise<void> { /* ... */ }
  private async unloadSubModules(context: ModuleContext): Promise<void> { /* ... */ }
}
```
