## 7.2 模块类实现

```typescript
/**
 * MES模块主类
 */
import { BaseModule, ModuleContext, ModuleStatus } from '@daoda/core';
import { ProductionModule } from './submodules/production';
import { MaterialModule } from './submodules/material';
import { QualityModule } from './submodules/quality';
import { EquipmentModule } from './submodules/equipment';
import { DataCollectionModule } from './submodules/collection';

export class MESModule extends BaseModule {
  readonly id = '@daoda/mes';
  readonly name = 'MES生产执行系统';
  readonly version = '1.0.0';
  
  private subModules: Map<string, BaseModule> = new Map();
  private dataCollectors: Map<string, DataCollector> = new Map();
  private scheduledTasks: Map<string, NodeJS.Timeout> = new Map();
  private wsConnections: Map<string, WebSocket> = new Map();
  
  protected _status: ModuleStatus = 'installed';
  
  async onInstall(context: ModuleContext): Promise<void> {
    context.logger.info(`Installing ${this.name}...`);
    
    await this.runMigrations('up');
    await this.initBaseData(context);
    await this.registerPermissions(context);
    
    context.logger.info(`${this.name} installed successfully`);
  }
  
  async onUninstall(context: ModuleContext): Promise<void> {
    context.logger.info(`Uninstalling ${this.name}...`);
    
    const hasData = await this.checkRelatedData(context);
    if (hasData) {
      throw new Error('模块存在关联数据，请先清理数据后再卸载');
    }
    
    await this.runMigrations('down');
    await this.unregisterPermissions(context);
    
    context.logger.info(`${this.name} uninstalled successfully`);
  }
  
  async onLoad(context: ModuleContext): Promise<void> {
    context.logger.info(`Loading ${this.name}...`);
    
    this.context = context;
    this._status = 'loaded';
    
    await this.registerServices(context);
    await this.loadConfig(context);
    await this.initSubModules(context);
    await this.initDataCollectors(context);
    
    context.logger.info(`${this.name} loaded successfully`);
  }
  
  async onUnload(context: ModuleContext): Promise<void> {
    context.logger.info(`Unloading ${this.name}...`);
    
    // 停止数据采集
    await this.stopDataCollectors();
    
    // 关闭WebSocket连接
    await this.closeWebSocketConnections();
    
    // 停止定时任务
    await this.stopScheduledTasks();
    
    await this.unloadSubModules(context);
    context.serviceRegistry.unregisterAll(this.id);
    
    this._status = 'installed';
    context.logger.info(`${this.name} unloaded successfully`);
  }
  
  async onEnable(context: ModuleContext): Promise<void> {
    context.logger.info(`Enabling ${this.name}...`);
    
    await this.registerRoutes(context);
    await this.registerMenus(context);
    await this.registerDashboardWidgets(context);
    await this.startEventSubscriptions(context);
    await this.enableSubModules(context);
    
    // 启动数据采集
    await this.startDataCollectors(context);
    
    // 启动WebSocket服务
    await this.startWebSocketServer(context);
    
    // 启动定时任务
    await this.startScheduledTasks(context);
    
    this._status = 'running';
    context.eventBus.emit('module.enabled', { moduleId: this.id });
    
    context.logger.info(`${this.name} enabled successfully`);
  }
  
  async onDisable(context: ModuleContext): Promise<void> {
    context.logger.info(`Disabling ${this.name}...`);
    
    await this.stopDataCollectors();
    await this.closeWebSocketConnections();
    await this.stopScheduledTasks();
    await this.disableSubModules(context);
    
    context.router.unregisterAll(this.id);
    context.menuRegistry.unregisterAll(this.id);
    context.dashboardRegistry.unregisterAll(this.id);
    context.eventBus.clearModuleSubscriptions(this.id);
    
    this._status = 'disabled';
    context.eventBus.emit('module.disabled', { moduleId: this.id });
    
    context.logger.info(`${this.name} disabled successfully`);
  }
  
  async onBeforeUpdate(context: ModuleContext, newVersion: string): Promise<void> {
    await this.saveModuleState(context);
    await this.backupModuleData(context);
    
    if (this._status === 'running') {
      await this.onDisable(context);
    }
    await this.onUnload(context);
  }
  
  async onAfterUpdate(context: ModuleContext, oldVersion: string): Promise<void> {
    await this.runMigrations('up');
    await this.restoreModuleState(context);
    await this.onLoad(context);
    await this.onEnable(context);
  }
  
  /**
   * 初始化数据采集器
   */
  private async initDataCollectors(context: ModuleContext): Promise<void> {
    const manifest = await this.getManifest();
    
    for (const collectorDef of manifest.dataCollectors || []) {
      if (!collectorDef.enabled) continue;
      
      let collector: DataCollector;
      switch (collectorDef.type) {
        case 'plc':
          collector = new PLCDataCollector(collectorDef.id, collectorDef.protocol);
          break;
        case 'rfid':
          collector = new RFIDDataCollector(collectorDef.id);
          break;
        case 'barcode':
          collector = new BarcodeDataCollector(collectorDef.id);
          break;
      }
      
      if (collector) {
        await collector.initialize();
        this.dataCollectors.set(collectorDef.id, collector);
      }
    }
  }
  
  /**
   * 启动数据采集
   */
  private async startDataCollectors(context: ModuleContext): Promise<void> {
    for (const [id, collector] of this.dataCollectors) {
      collector.on('data', (data: CollectedData) => {
        this.handleCollectedData(context, id, data);
      });
      
      collector.on('error', (error: Error) => {
        context.logger.error(`Data collector ${id} error`, error);
      });
      
      await collector.start();
      context.logger.info(`Data collector ${id} started`);
    }
  }
  
  /**
   * 停止数据采集
   */
  private async stopDataCollectors(): Promise<void> {
    for (const [id, collector] of this.dataCollectors) {
      await collector.stop();
      this.context?.logger?.info(`Data collector ${id} stopped`);
    }
  }
  
  /**
   * 处理采集的数据
   */
  private async handleCollectedData(
    context: ModuleContext, 
    collectorId: string, 
    data: CollectedData
  ): Promise<void> {
    // 根据采集器类型处理数据
    switch (collectorId) {
      case 'plc-collector':
        await this.processPLCData(context, data);
        break;
      case 'rfid-collector':
        await this.processRFIDData(context, data);
        break;
      case 'barcode-collector':
        await this.processBarcodeData(context, data);
        break;
    }
  }
  
  /**
   * 处理PLC数据
   */
  private async processPLCData(context: ModuleContext, data: CollectedData): Promise<void> {
    const equipmentService = context.serviceRegistry.get<EquipmentService>('EquipmentService');
    
    // 更新设备状态和参数
    for (const [param, value] of Object.entries(data.values)) {
      await equipmentService.updateParameter(data.equipmentId, param, value);
    }
    
    // 检查告警
    const alarms = await equipmentService.checkAlarms(data.equipmentId, data.values);
    for (const alarm of alarms) {
      context.eventBus.emit('mes.equipment.alarm', {
        equipmentId: data.equipmentId,
        alarmCode: alarm.code,
        alarmLevel: alarm.level,
        message: alarm.message,
        timestamp: Date.now()
      });
    }
    
    // 推送到WebSocket客户端
    this.broadcastToWebSocket('equipment.parameters.' + data.equipmentId, {
      equipmentId: data.equipmentId,
      parameters: data.values,
      timestamp: data.timestamp
    });
  }
  
  /**
   * 启动WebSocket服务器
   */
  private async startWebSocketServer(context: ModuleContext): Promise<void> {
    // WebSocket服务通常由核心系统提供，这里注册消息处理器
    context.wsServer?.registerHandler('mes.subscribe', async (client, message) => {
      const { topics } = message;
      
      for (const topic of topics) {
        client.subscribe(topic);
      }
      
      return { success: true, subscribed: topics };
    });
    
    context.wsServer?.registerHandler('mes.unsubscribe', async (client, message) => {
      const { topics } = message;
      
      for (const topic of topics) {
        client.unsubscribe(topic);
      }
      
      return { success: true };
    });
  }
  
  /**
   * 广播WebSocket消息
   */
  private broadcastToWebSocket(topic: string, data: any): void {
    this.context?.wsServer?.broadcast(topic, {
      topic,
      timestamp: Date.now(),
      data
    });
  }
```

```typescript
 * 支持设备监控、维护管理、OEE计算
 */
export class EquipmentModule extends BaseModule {
  readonly id = '@daoda/mes.equipment';
  readonly name = '设备管理';
  readonly version = '1.0.0';
  
  private healthCheckTimer?: NodeJS.Timeout;
  
  async onEnable(context: ModuleContext): Promise<void> {
    // 注册路由
    context.router.register('GET', '/api/v1/mes/equipment', 
      EquipmentController.list, this.id);
    context.router.register('GET', '/api/v1/mes/equipment/:id', 
      EquipmentController.detail, this.id);
    context.router.register('GET', '/api/v1/mes/equipment/:id/status', 
      EquipmentController.status, this.id);
    context.router.register('GET', '/api/v1/mes/equipment/:id/oee', 
      EquipmentController.oee, this.id);
    context.router.register('POST', '/api/v1/mes/equipment/:id/repair', 
      EquipmentController.reportRepair, this.id);
    
    // 启动健康检查
    this.startHealthCheck(context);
    
    // 订阅设备告警事件
    context.eventBus.on('mes.equipment.alarm', 
      this.handleEquipmentAlarm.bind(this), this.id);
  }
  
  async onDisable(context: ModuleContext): Promise<void> {
    if (this.healthCheckTimer) {
      clearInterval(this.healthCheckTimer);
    }
    
    context.router.unregisterAll(this.id);
    context.eventBus.clearModuleSubscriptions(this.id);
  }
  
  /**
   * 启动设备健康检查
   */
  private startHealthCheck(context: ModuleContext): void {
    this.healthCheckTimer = setInterval(async () => {
      const equipmentService = context.serviceRegistry.get<EquipmentService>('EquipmentService');
      
      // 检查设备连接状态
      const equipments = await equipmentService.getAllActive();
      
      for (const equipment of equipments) {
        const isHealthy = await this.checkEquipmentHealth(equipment);
        
        if (!isHealthy && equipment.status !== 'offline') {
          // 设备离线
          await equipmentService.updateStatus(equipment.id, 'offline');
          
          context.eventBus.emit('mes.equipment.status_changed', {
            equipmentId: equipment.id,
            oldStatus: equipment.status,
            newStatus: 'offline',
            reason: 'health_check_failed'
          });
        }
      }
    }, 5 * 60 * 1000); // 每5分钟
  }
  
  /**
   * 处理设备告警
   */
  private async handleEquipmentAlarm(event: any): Promise<void> {
    const { equipmentId, alarmCode, alarmLevel, message } = event.payload;
    
    const notificationService = this.context.serviceRegistry.get<NotificationService>('NotificationService');
    
    // 根据告警级别发送通知
    if (alarmLevel === 'critical' || alarmLevel === 'high') {
      await notificationService.send({
        type: 'equipment_alarm',
        title: `设备告警 - ${alarmLevel}`,
        content: `设备 ${equipmentId}: ${message}`,
        recipients: 'maintenance_team',
        priority: 'high'
      });
    }
    
    // 记录告警日志
    const equipmentService = this.context.serviceRegistry.get<EquipmentService>('EquipmentService');
    await equipmentService.recordAlarm(equipmentId, {
      code: alarmCode,
      level: alarmLevel,
      message,
      timestamp: Date.now()
    });
  }
  
  private async checkEquipmentHealth(equipment: Equipment): Promise<boolean> {
    // 检查设备最近是否上报数据
    const lastReportTime = equipment.lastReportTime;
    const now = Date.now();
    
    // 超过5分钟未上报视为离线
    if (lastReportTime && (now - lastReportTime) > 5 * 60 * 1000) {
      return false;
    }
    
    return true;
  }
}

/**
 * OEE服务
 * 计算设备综合效率
 */
class OEEService {
  constructor(private context: ModuleContext) {}
  
  /**
   * 计算OEE
   */
  async calculate(equipmentId: string, period: TimePeriod): Promise<OEE> {
    const equipmentService = this.context.serviceRegistry.get<EquipmentService>('EquipmentService');
    
    // 获取时间范围内的运行数据
    const runData = await equipmentService.getRunData(equipmentId, period);
    
    // 计算可用率 (Availability)
    const availability = this.calculateAvailability(runData);
    
    // 计算性能率 (Performance)
    const performance = this.calculatePerformance(runData);
    
    // 计算良品率 (Quality)
    const quality = this.calculateQuality(runData);
    
    // 计算OEE
    const oee = availability * performance * quality;
    
    // 保存OEE记录
    await equipmentService.saveOEE(equipmentId, period, {
      availability,
      performance,
      quality,
      oee
    });
    
    return {
      equipmentId,
      period,
      availability,
      performance,
      quality,
      oee
    };
  }
  
  /**
   * 计算可用率
   * 可用率 = 运行时间 / 计划生产时间
   */
  private calculateAvailability(data: RunData): number {
    const { plannedTime, runTime, downtime } = data;
    
    // 可用率 = (计划时间 - 停机时间) / 计划时间
    const availability = (plannedTime - downtime) / plannedTime;
    
    return Math.max(0, Math.min(1, availability));
  }
  
  /**
   * 计算性能率
   * 性能率 = 实际产量 / 理论产量
   */
  private calculatePerformance(data: RunData): number {
    const { actualOutput, idealOutput } = data;
    
    // 性能率 = 实际产量 / 理论最大产量
    const performance = actualOutput / idealOutput;
    
    return Math.max(0, Math.min(1, performance));
  }
  
  /**
   * 计算良品率
   * 良品率 = 合格品数量 / 总产量
   */
  private calculateQuality(data: RunData): number {
    const { goodOutput, totalOutput } = data;
    
    // 良品率 = 合格品 / 总产量
    const quality = goodOutput / totalOutput;
    
    return Math.max(0, Math.min(1, quality));
  }
}

interface OEE {
  equipmentId: string;
  period: TimePeriod;
  availability: number;
  performance: number;
  quality: number;
  oee: number;
}

interface TimePeriod {
  start: Date;
  end: Date;
}
```
