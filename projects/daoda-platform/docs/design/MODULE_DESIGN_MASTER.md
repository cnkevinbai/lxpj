## 四、模块核心接口定义

### 4.1 模块基类

```typescript
/**
 * 模块基类
 * 所有业务模块必须继承此类
 */
abstract class BaseModule implements ModuleLifecycleHooks {
  // ============ 元数据 ============
  abstract readonly id: string;
  abstract readonly name: string;
  abstract readonly version: string;
  
  // ============ 上下文 ============
  protected context: ModuleContext;
  
  // ============ 状态 ============
  protected _status: ModuleStatus = 'installed';
  
  // ============ 注册表 ============
  protected controllers: Map<string, any> = new Map();
  protected services: Map<string, any> = new Map();
  protected repositories: Map<string, any> = new Map();
  protected events: Map<string, any> = new Map();
  protected routes: Map<string, any> = new Map();
  
  // ============ 生命周期钩子 ============
  async onInstall(context: ModuleContext): Promise<void> {
    // 默认实现：运行数据库迁移
    await this.runMigrations('up');
  }
  
  async onUninstall(context: ModuleContext): Promise<void> {
    // 默认实现：回滚数据库迁移
    await this.runMigrations('down');
  }
  
  async onLoad(context: ModuleContext): Promise<void> {
    this.context = context;
    this._status = 'loaded';
    
    // 注册服务
    await this.registerServices();
    
    // 注册事件
    await this.registerEvents();
  }
  
  async onUnload(context: ModuleContext): Promise<void> {
    // 清理资源
    this.controllers.clear();
    this.services.clear();
    this.repositories.clear();
    this.events.clear();
    this.routes.clear();
    
    this._status = 'installed';
  }
  
  async onEnable(context: ModuleContext): Promise<void> {
    // 注册路由
    await this.registerRoutes();
    
    // 启动事件订阅
    await this.startEventSubscriptions();
    
    this._status = 'running';
    
    // 发布模块启用事件
    context.eventBus.emit('module.enabled', { moduleId: this.id });
  }
  
  async onDisable(context: ModuleContext): Promise<void> {
    // 注销路由
    await this.unregisterRoutes();
    
    // 停止事件订阅
    await this.stopEventSubscriptions();
    
    this._status = 'disabled';
    
    // 发布模块禁用事件
    context.eventBus.emit('module.disabled', { moduleId: this.id });
  }
  
  async onBeforeUpdate(context: ModuleContext, newVersion: string): Promise<void> {
    // 保存模块状态
    await this.saveState();
    
    // 备份数据
    await this.backupData();
  }
  
  async onAfterUpdate(context: ModuleContext, oldVersion: string): Promise<void> {
    // 运行数据迁移
    await this.runMigrations('up');
    
    // 恢复模块状态
    await this.restoreState();
  }
  
  // ============ 辅助方法 ============
  protected async registerServices(): Promise<void> {
    for (const [name, ServiceClass] of this.services) {
      this.context.serviceRegistry.register(name, new ServiceClass(this.context));
    }
  }
  
  protected async registerRoutes(): Promise<void> {
    for (const [path, handler] of this.routes) {
      this.context.router.register(path, handler);
    }
  }
  
  protected async runMigrations(direction: 'up' | 'down'): Promise<void> {
    await this.context.migrationRunner.run(this.id, direction);
  }
  
  // ============ 状态访问器 ============
  get status(): ModuleStatus {
    return this._status;
  }
}

// 模块状态类型
type ModuleStatus = 
  | 'installed'    // 已安装
  | 'loaded'       // 已加载
  | 'running'      // 运行中
  | 'disabled'     // 已禁用
  | 'updating'     // 更新中
  | 'error';       // 错误
```

```typescript
/**
 * 模块加载器
 * 负责模块的加载、卸载、启用、禁用
 */
class ModuleLoader {
  private modules: Map<string, LoadedModule> = new Map();
  private context: ModuleContext;
  
  /**
   * 加载模块
   */
  async loadModule(moduleId: string): Promise<void> {
    // 1. 检查模块是否已加载
    if (this.modules.has(moduleId)) {
      throw new Error(`Module ${moduleId} is already loaded`);
    }
    
    // 2. 获取模块清单
    const manifest = await this.getModuleManifest(moduleId);
    
    // 3. 检查依赖
    await this.checkDependencies(manifest);
    
    // 4. 创建模块上下文
    const moduleContext = this.createModuleContext(manifest);
    
    // 5. 加载模块代码
    const ModuleClass = await this.importModule(moduleId);
    const moduleInstance = new ModuleClass();
    
    // 6. 调用加载钩子
    await moduleInstance.onLoad(moduleContext);
    
    // 7. 注册模块
    this.modules.set(moduleId, {
      manifest,
      instance: moduleInstance,
      context: moduleContext,
      status: 'loaded'
    });
    
    this.logger.info(`Module ${moduleId} loaded successfully`);
  }
  
  /**
   * 卸载模块
   */
  async unloadModule(moduleId: string): Promise<void> {
    const loadedModule = this.modules.get(moduleId);
    if (!loadedModule) {
      throw new Error(`Module ${moduleId} is not loaded`);
    }
    
    // 1. 先禁用模块
    if (loadedModule.status === 'running') {
      await this.disableModule(moduleId);
    }
    
    // 2. 调用卸载钩子
    await loadedModule.instance.onUnload(loadedModule.context);
    
    // 3. 清理资源
    loadedModule.context.serviceRegistry.unregisterAll(moduleId);
    loadedModule.context.router.unregisterAll(moduleId);
    
    // 4. 移除模块
    this.modules.delete(moduleId);
    
    // 5. 清理缓存
    this.clearModuleCache(moduleId);
    
    this.logger.info(`Module ${moduleId} unloaded successfully`);
  }
  
  /**
   * 启用模块
   */
  async enableModule(moduleId: string): Promise<void> {
    const loadedModule = this.modules.get(moduleId);
    if (!loadedModule) {
      throw new Error(`Module ${moduleId} is not loaded`);
    }
    
    if (loadedModule.status === 'running') {
      return; // 已经在运行
    }
    
    // 调用启用钩子
    await loadedModule.instance.onEnable(loadedModule.context);
    
    loadedModule.status = 'running';
    
    this.logger.info(`Module ${moduleId} enabled successfully`);
  }
  
  /**
   * 禁用模块
   */
  async disableModule(moduleId: string): Promise<void> {
    const loadedModule = this.modules.get(moduleId);
    if (!loadedModule) {
      throw new Error(`Module ${moduleId} is not loaded`);
    }
    
    if (loadedModule.status !== 'running') {
      return; // 不在运行状态
    }
    
    // 调用禁用钩子
    await loadedModule.instance.onDisable(loadedModule.context);
    
    loadedModule.status = 'disabled';
    
    this.logger.info(`Module ${moduleId} disabled successfully`);
  }
  
  /**
   * 检查依赖
   */
  private async checkDependencies(manifest: ModuleManifest): Promise<void> {
    // 检查核心版本
    if (manifest.dependencies.core) {
      const coreVersion = this.context.coreVersion;
      if (!semver.satisfies(coreVersion, manifest.dependencies.core)) {
        throw new Error(
          `Module ${manifest.id} requires core version ${manifest.dependencies.core}, ` +
          `but current version is ${coreVersion}`
        );
      }
    }
    
    // 检查模块依赖
    for (const [depId, version] of Object.entries(manifest.dependencies.modules || {})) {
      const loadedDep = this.modules.get(depId);
      if (!loadedDep) {
        throw new Error(`Module ${manifest.id} depends on ${depId}, which is not loaded`);
      }
      if (!semver.satisfies(loadedDep.manifest.version, version)) {
        throw new Error(
          `Module ${manifest.id} requires ${depId}@${version}, ` +
          `but ${depId}@${loadedDep.manifest.version} is loaded`
        );
      }
    }
    
    // 检查冲突
    for (const conflictId of manifest.conflicts || []) {
      if (this.modules.has(conflictId)) {
        throw new Error(`Module ${manifest.id} conflicts with ${conflictId}`);
      }
    }
  }
}

interface LoadedModule {
  manifest: ModuleManifest;
  instance: BaseModule;
  context: ModuleContext;
  status: ModuleStatus;
}
```

```typescript
/**
 * 模块热更新器
 * 支持模块的热更新，无需重启系统
 */
class ModuleHotUpdater {
  private moduleLoader: ModuleLoader;
  private eventBus: EventBus;
  
  /**
   * 热更新模块
   */
  async updateModule(moduleId: string, newVersion: string): Promise<void> {
    const loadedModule = this.moduleLoader.getModule(moduleId);
    if (!loadedModule) {
      throw new Error(`Module ${moduleId} is not loaded`);
    }
    
    const oldVersion = loadedModule.manifest.version;
    const manifest = loadedModule.manifest;
    
    // 1. 检查是否支持热更新
    if (!manifest.hotUpdate.enabled) {
      throw new Error(`Module ${moduleId} does not support hot update`);
    }
    
    // 2. 根据更新策略执行
    const strategy = manifest.hotUpdate.strategy;
    
    switch (strategy) {
      case 'rolling':
        await this.rollingUpdate(moduleId, newVersion);
        break;
      case 'blue-green':
        await this.blueGreenUpdate(moduleId, newVersion);
        break;
      case 'canary':
        await this.canaryUpdate(moduleId, newVersion);
        break;
    }
    
    // 3. 发布更新事件
    this.eventBus.emit('module.updated', {
      moduleId,
      oldVersion,
      newVersion,
      timestamp: Date.now()
    });
  }
  
  /**
   * 滚动更新
   * 逐个实例更新，保证服务不中断
   */
  private async rollingUpdate(moduleId: string, newVersion: string): Promise<void> {
    const loadedModule = this.moduleLoader.getModule(moduleId);
    
    // 1. 更新前钩子
    await loadedModule.instance.onBeforeUpdate(loadedModule.context, newVersion);
    
    // 2. 禁用模块
    await this.moduleLoader.disableModule(moduleId);
    
    // 3. 备份当前状态
    const stateBackup = await this.backupModuleState(moduleId);
    
    try {
      // 4. 加载新版本代码
      await this.loadNewVersion(moduleId, newVersion);
      
      // 5. 启用新版本
      await this.moduleLoader.enableModule(moduleId);
      
      // 6. 更新后钩子
      await loadedModule.instance.onAfterUpdate(loadedModule.context, 
        loadedModule.manifest.version);
      
      // 7. 验证新版本
      await this.verifyModule(moduleId);
      
      this.logger.info(`Module ${moduleId} updated to ${newVersion} (rolling)`);
      
    } catch (error) {
      // 回滚
      await this.rollbackModule(moduleId, stateBackup);
      throw error;
    }
  }
  
  /**
   * 蓝绿更新
   * 新旧版本并行运行，完成后切换
   */
  private async blueGreenUpdate(moduleId: string, newVersion: string): Promise<void> {
    // 1. 启动新版本（蓝）
    const newModuleId = `${moduleId}-new`;
    await this.moduleLoader.loadModule(newModuleId);
    await this.moduleLoader.enableModule(newModuleId);
    
    try {
      // 2. 验证新版本
      await this.verifyModule(newModuleId);
      
      // 3. 切换流量到新版本
      await this.switchTraffic(moduleId, newModuleId);
      
      // 4. 停止旧版本（绿）
      await this.moduleLoader.disableModule(moduleId);
      await this.moduleLoader.unloadModule(moduleId);
      
      // 5. 重命名新版本
      await this.renameModule(newModuleId, moduleId);
      
      this.logger.info(`Module ${moduleId} updated to ${newVersion} (blue-green)`);
      
    } catch (error) {
      // 回滚到旧版本
      await this.moduleLoader.disableModule(newModuleId);
      await this.moduleLoader.unloadModule(newModuleId);
      throw error;
    }
  }
  
  /**
   * 金丝雀更新
   * 逐步放量，先让部分用户使用新版本
   */
  private async canaryUpdate(moduleId: string, newVersion: string): Promise<void> {
    const canaryStages = [5, 10, 25, 50, 100]; // 放量比例
    
    // 1. 启动新版本
    const newModuleId = `${moduleId}-canary`;
    await this.moduleLoader.loadModule(newModuleId);
    await this.moduleLoader.enableModule(newModuleId);
    
    for (const percentage of canaryStages) {
      // 2. 按比例分配流量
      await this.distributeTraffic(moduleId, newModuleId, percentage);
      
      // 3. 观察指标
      await this.observeMetrics(newModuleId, 60000); // 观察1分钟
      
      // 4. 检查是否有错误
      const errorRate = await this.getErrorRate(newModuleId);
      if (errorRate > 0.01) { // 错误率超过1%
        // 回滚
        await this.moduleLoader.disableModule(newModuleId);
        await this.moduleLoader.unloadModule(newModuleId);
        throw new Error(`Canary update failed at ${percentage}%: error rate ${errorRate}`);
      }
    }
    
    // 5. 完成更新
    await this.moduleLoader.disableModule(moduleId);
    await this.moduleLoader.unloadModule(moduleId);
    await this.renameModule(newModuleId, moduleId);
    
    this.logger.info(`Module ${moduleId} updated to ${newVersion} (canary)`);
  }
  
  /**
   * 验证模块
   */
  private async verifyModule(moduleId: string): Promise<boolean> {
    // 执行健康检查
    const health = await this.healthCheck(moduleId);
    if (!health.healthy) {
      throw new Error(`Module ${moduleId} health check failed: ${health.message}`);
    }
    
    // 执行集成测试
    const testResult = await this.runIntegrationTests(moduleId);
    if (!testResult.passed) {
      throw new Error(`Module ${moduleId} integration tests failed`);
    }
    
    return true;
  }
}
```

```typescript
/**
 * 事件总线
 * 模块间异步通信的核心机制
 */
class EventBus {
  private handlers: Map<string, Set<EventHandler>> = new Map();
  private moduleSubscriptions: Map<string, string[]> = new Map();
  
  /**
   * 发布事件
   */
  emit(event: string, payload: any, options?: EmitOptions): void {
    const handlers = this.handlers.get(event);
    if (!handlers || handlers.size === 0) return;
    
    for (const handler of handlers) {
      try {
        // 异步执行
        setImmediate(() => {
          handler({
            event,
            payload,
            timestamp: Date.now(),
            ...options
          });
        });
      } catch (error) {
        this.logger.error(`Event handler error: ${event}`, error);
      }
    }
  }
  
  /**
   * 订阅事件
   */
  on(event: string, handler: EventHandler, moduleId?: string): void {
    if (!this.handlers.has(event)) {
      this.handlers.set(event, new Set());
    }
    this.handlers.get(event)!.add(handler);
    
    // 记录模块订阅，便于卸载时清理
    if (moduleId) {
      if (!this.moduleSubscriptions.has(moduleId)) {
        this.moduleSubscriptions.set(moduleId, []);
      }
      this.moduleSubscriptions.get(moduleId)!.push(event);
    }
  }
  
  /**
   * 取消订阅
   */
  off(event: string, handler: EventHandler): void {
    const handlers = this.handlers.get(event);
    if (handlers) {
      handlers.delete(handler);
    }
  }
  
  /**
   * 清理模块的所有订阅
   */
  clearModuleSubscriptions(moduleId: string): void {
    const events = this.moduleSubscriptions.get(moduleId);
    if (!events) return;
    
    for (const event of events) {
      const handlers = this.handlers.get(event);
      if (handlers) {
        // 移除该模块注册的所有处理器
        for (const handler of handlers) {
          if ((handler as any)._moduleId === moduleId) {
            handlers.delete(handler);
          }
        }
      }
    }
    
    this.moduleSubscriptions.delete(moduleId);
  }
}

interface EventContext {
  event: string;
  payload: any;
  timestamp: number;
  sourceModule?: string;
  correlationId?: string;
}

type EventHandler = (context: EventContext) => void | Promise<void>;
```

```typescript
/**
 * 服务注册表
 * 模块间服务调用机制
 */
class ServiceRegistry {
  private services: Map<string, ServiceDescriptor> = new Map();
  private moduleServices: Map<string, string[]> = new Map();
  
  /**
   * 注册服务
   */
  register(name: string, service: any, moduleId?: string): void {
    if (this.services.has(name)) {
      throw new Error(`Service ${name} is already registered`);
    }
    
    this.services.set(name, {
      name,
      instance: service,
      moduleId,
      registeredAt: Date.now()
    });
    
    if (moduleId) {
      if (!this.moduleServices.has(moduleId)) {
        this.moduleServices.set(moduleId, []);
      }
      this.moduleServices.get(moduleId)!.push(name);
    }
  }
  
  /**
   * 获取服务
   */
  get<T>(name: string): T {
    const descriptor = this.services.get(name);
    if (!descriptor) {
      throw new Error(`Service ${name} not found`);
    }
    return descriptor.instance as T;
  }
  
  /**
   * 检查服务是否存在
   */
  has(name: string): boolean {
    return this.services.has(name);
  }
  
  /**
   * 注销服务
   */
  unregister(name: string): void {
    const descriptor = this.services.get(name);
    if (descriptor) {
      this.services.delete(name);
      
      // 从模块服务列表中移除
      if (descriptor.moduleId) {
        const moduleServices = this.moduleServices.get(descriptor.moduleId);
        if (moduleServices) {
          const index = moduleServices.indexOf(name);
          if (index > -1) {
            moduleServices.splice(index, 1);
          }
        }
      }
    }
  }
  
  /**
   * 注销模块的所有服务
   */
  unregisterAll(moduleId: string): void {
    const services = this.moduleServices.get(moduleId);
    if (!services) return;
    
    for (const name of services) {
      this.services.delete(name);
    }
    
    this.moduleServices.delete(moduleId);
  }
}

interface ServiceDescriptor {
  name: string;
  instance: any;
  moduleId?: string;
  registeredAt: number;
}
```

### 9.2 模块数据隔离

```typescript
/**
 * 模块数据隔离策略
 */
class ModuleDataIsolation {
  /**
   * 检查数据访问权限
   */
  async checkDataAccess(
    moduleId: string,
    userId: string,
    resource: string,
    resourceId: string,
    action: string
  ): Promise<boolean> {
    // 1. 获取用户权限
    const permissions = await this.getUserPermissions(userId);
    
    // 2. 检查模块权限
    const modulePermission = `${moduleId}:${resource}:${action}`;
    if (!permissions.includes(modulePermission)) {
      return false;
    }
    
    // 3. 检查数据范围
    const dataScope = await this.getDataScope(userId, modulePermission);
    if (dataScope === 'all') {
      return true;
    }
    
    // 4. 检查数据归属
    const resourceOwnerId = await this.getResourceOwnerId(resource, resourceId);
    if (dataScope === 'self' && resourceOwnerId !== userId) {
      return false;
    }
    
    if (dataScope === 'dept') {
      const userDeptId = await this.getUserDeptId(userId);
      const resourceDeptId = await this.getResourceDeptId(resource, resourceId);
      return userDeptId === resourceDeptId;
    }
    
    return true;
  }
}
```

### 5.2 数据同步一致性

| 数据类型 | 一致性策略 | 冲突解决 |
|----------|------------|----------|
| 用户数据 | 强一致性 | 最后写入优先 |
| 订单数据 | 最终一致性 | 来源优先（门户>APP） |
| 库存数据 | 强一致性 | 最后写入优先 |
| 审批数据 | 因果一致性 | 来源优先 |

### 5.3 离线数据同步（APP端）

```
【离线保存流程】
APP保存数据 ──> 检查网络状态
             ├── 在线: 直接同步服务器 → 成功则本地标记synced
             │              └── 失败则加入pending队列
             └── 离线: 保存到本地数据库，标记pending

【网络恢复同步】
网络恢复 ──> 检查pending队列
          ──> 逐条同步到服务器
          ──> 处理冲突（服务器优先/合并）
          ──> 更新本地状态
```

## 六、API网关路由配置

### 6.1 路由规则

| 路由前缀 | 目标服务 | 认证要求 | 限流 |
|----------|----------|----------|------|
| `/api/v1/website` | website-server | 无需认证 | 100/分钟 |
| `/api/v1/portal` | portal-server | JWT认证 | 1000/分钟 |
| `/api/v1/mobile` | mobile-gateway | JWT + 设备认证 | 500/分钟 |
| `/api/v1/auth` | auth-service | 无需认证 | 50/分钟 |
| `/api/v1/crm` | crm-service | JWT + 模块权限 | 1000/分钟 |
| `/api/v1/erp` | erp-service | JWT + 模块权限 | 1000/分钟 |
| `/api/v1/service` | service-module | JWT + 模块权限 | 1000/分钟 |
| `/api/v1/workflow` | workflow-service | JWT + 模块权限 | 1000/分钟 |
| `/ws` | websocket-gateway | JWT (query参数) | - |

### 6.2 认证授权流程

```
【官网访问】
无需认证 → 公开API（产品列表、新闻等）
         → 有用户行为则可选登录（询价、下单等）

【门户访问】
用户登录 → JWT Token → API网关验证 → 权限检查 → 访问资源

【APP访问】
用户登录 → JWT Token + DeviceId → API网关验证 → 设备绑定检查 → 权限检查 → 访问资源
```
